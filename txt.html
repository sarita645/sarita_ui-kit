// 變數定義
$border-style: solid;
$border-width: 0px;
$padding: 0px;
$display: flex;
$align-items: center;
$justify-content: center;
$margin: 2px 0;
$line-height: 1.5;
$default-font-size: 16px;
$font-family: "NotoSansTc-Regular", sans-serif;
$color: #333333;

// 混合，用於生成文字樣式
@mixin text-style($font-size: $default-font-size, $font-weight: normal) {
border-style: $border-style;
border-width: $border-width;
font-size: $font-size;
font-weight: $font-weight;
padding: $padding;
// display: $display;
// align-items: $align-items;
// justify-content: $justify-content;
margin: $margin;
line-height: $line-height;
font-family: $font-family;
color: $color;
}

// 具體的樣式
.txt-8 {
@include text-style(8px);
}

.txt-8-B {
@include text-style(8px, bold);
}

.txt-12 {
@include text-style(12px);
}

.txt-12-B {
@include text-style(12px, bold);
}

.txt-16 {
@include text-style(16px);
}

.txt-16-B {
@include text-style(16px, bold);
}
